function J = computeLogisticRegressionCost(X, y, w, b)
% Arguments:
% X (m,n) : Data, m examples with n features
% y (m,1) : Target values
% w (n,1) : model weights
% b (1,1) : model bias
m = size(X,1);
fwb = zeros(m,1);
J_tmp = 0;
for i = 1:m
z = X(i,:)*w + b;
fwb(i) = ( 1/(1+exp(-z)) );
J_tmp = J_tmp + ( (1-y(i))*log(1-fwb(i)) + y(i)*log(fwb(i)) );
end
J = -(1/m)*J_tmp;
end
